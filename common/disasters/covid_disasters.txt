covid_pandemic = {
    potential = {
        has_country_flag = covid_alerted
    }

    can_start = yes

    start_trigger = {
        has_country_flag = covid_alerted
    }

    progress = {
        base = 2

        modifier = {
            factor = 1.5
            num_of_provinces_with = {
                limit = { has_province_flag = infected_province }
                count >= 5
            }
        }

        modifier = {
            factor = 2.0
            num_of_provinces_with = {
                limit = { has_province_flag = infected_province }
                count >= 10
            }
        }
    }

    on_start = {
        add_stability = -1
        add_corruption = 2
    }

    on_end = {
        clr_country_flag = covid_alerted
        remove_country_modifier = covid_quarantine_nationwide
        remove_country_modifier = covid_disaster_penalty
    }

    monthly_action = {
        add_country_modifier = {
            name = covid_disaster_penalty
            duration = 30
        }
    }

    effect = {
        add_reform_progress = -10
        add_corruption = 0.5
    }

    ai_will_do = {
        factor = 1
    }
}
